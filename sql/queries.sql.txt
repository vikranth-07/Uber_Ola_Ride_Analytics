-- USE DATABASE
USE uber_analytics;

-- TOTAL REVENUE
SELECT SUM(fare_amount) AS total_revenue
FROM rides;

-- TOTAL TRIPS
SELECT COUNT(*) AS total_trips
FROM rides;

-- PEAK BOOKING HOURS
SELECT hour,
COUNT(*) AS total_rides
FROM rides
GROUP BY hour
ORDER BY total_rides DESC;

-- VEHICLE TYPE ANALYSIS
SELECT vehicle_type,
COUNT(*) AS total_rides
FROM rides
GROUP BY vehicle_type;

-- RIDE STATUS ANALYSIS
SELECT ride_status,
COUNT(*) AS total_rides
FROM rides
GROUP BY ride_status;

-- AVERAGE DRIVER RATING
SELECT AVG(driver_rating) AS avg_rating
FROM rides;

-- REVENUE BY VEHICLE TYPE
SELECT vehicle_type,
SUM(fare_amount) AS revenue
FROM rides
GROUP BY vehicle_type
ORDER BY revenue DESC;

-- PAYMENT METHOD ANALYSIS
SELECT payment_method,
COUNT(*) AS total_transactions
FROM rides
GROUP BY payment_method;

-- MONTHLY RIDE TRENDS
SELECT month,
COUNT(*) AS total_rides
FROM rides
GROUP BY month;

-- BEST UBER BASE
SELECT uber_base,
COUNT(*) AS total_rides
FROM rides
GROUP BY uber_base
ORDER BY total_rides DESC;
